```swift
class Solution {
    func jump(_ nums: [Int]) -> Int {
        var answer = 0
        var maxJumpIndex = 0
        var jumpedIndex = 0
        
        for i in 0..<nums.count - 1 {
            maxJumpIndex = max(maxJumpIndex, i + nums[i])
            
            if i == jumpedIndex {
                jumpedIndex = maxJumpIndex
                answer += 1
            }
        }
        return answer
    }
}
```
