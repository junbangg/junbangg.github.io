func isDividable(by number: Int, in numbers: [Int], M: Int) -> Bool {
    var _min = Int.max
    var _max = -Int.max
    var count = 1
    
    for i in 0..<numbers.count {
        _min = min(_min, numbers[i])
        _max = max(_max, numbers[i])
        if _max - _min > number {
            count += 1
            _max = numbers[i]
            _min = numbers[i]
        }
    }
    return count <= M
}
